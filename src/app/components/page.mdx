## Code Blocks

### TypeScript Example

```typescript showLineNumbers {3-5}
interface User {
  id: string;
  name: string;
  email: string;
  role: 'admin' | 'user';
}

function createUser(data: Omit<User, 'id'>): User {
  return {
    id: crypto.randomUUID(),
    ...data
  };
}

const newUser = createUser({
  name: "John Doe",
  email: "john@example.com",
  role: "user"
});
```
